```php
<?php
function getVisitorInfo() {
    $visitor = [];
    
    // جمع معلومات الزائر
    $visitor['ip'] = $_SERVER['REMOTE_ADDR'];
    $visitor['name'] = ''; // يمكن الحصول عليه من نموذج تسجيل الدخول
    $visitor['browser'] = $_SERVER['HTTP_USER_AGENT'];
    $visitor['page'] = $_SERVER['REQUEST_URI'];
    $visitor['time'] = date('Y-m-d H:i:s');
    
    // الحصول على معلومات الموقع الجغرافي
    $ip_details = json_decode(file_get_contents("http://ipinfo.io/{$visitor['ip']}/json"));
    $visitor['country'] = $ip_details->country ?? 'Unknown';
    $visitor['city'] = $ip_details->city ?? 'Unknown';
    
    return $visitor;
}

function saveVisitorInfo($conn, $visitor) {
    $stmt = $conn->prepare("INSERT INTO visitors 
        (ip_address, name, country, city, browser, visit_time, page_visited) 
        VALUES (?, ?, ?, ?, ?, ?, ?)");
    
    $stmt->bind_param("sssssss", 
        $visitor['ip'],
        $visitor['name'],
        $visitor['country'],
        $visitor['city'],
        $visitor['browser'],
        $visitor['time'],
        $visitor['page']);
    
    $stmt->execute();
}

// استخدام الدوال
$conn = new mysqli("localhost", "username", "password", "database_name");
$visitor = getVisitorInfo();
saveVisitorInfo($conn, $visitor);
?>
```
